Key Concepts

UI Controls in p5.js

p5.js provides built-in UI controls that make it easy to create interactive interfaces for your creative coding projects. These controls automatically handle user interaction, allowing you to focus on connecting them to your code. UI controls are essential for creating interactive installations, performances, and user-driven art.

Why Use UI Controls:

  • User-Friendly: Provide intuitive interfaces for non-programmers
  • Real-Time Control: Allow live manipulation of parameters
  • Experimentation: Enable exploration of different values quickly
  • Performance: Create interactive installations and live performances

Available UI Controls:

  • Sliders: For continuous values (size, position, color, etc.)
  • Buttons: For discrete actions (start/stop, reset, change mode)
  • Checkboxes: For boolean toggles (show/hide, enable/disable)
  • Select Menus: For choosing from multiple options

Creating UI Controls

Important: UI controls are created in setup(), not draw(). They are created once and persist throughout your sketch.

Sliders:

Code example 1 written in javascript. The code follows.
let sizeSlider;

function setup() {
  createCanvas(400, 400);
  
  // createSlider(min, max, [value], [step])
  // min: minimum value
  // max: maximum value
  // value: starting value (optional, defaults to middle)
  // step: increment size (optional, defaults to 1)
  sizeSlider = createSlider(10, 200, 50, 1);
  sizeSlider.position(20, 20);  // Position on screen
  sizeSlider.style('width', '200px');  // Set width
}

function draw() {
  background(220);
  
  // Read slider value
  let circleSize = sizeSlider.value();  // Returns number

  ellipse(200, 200, circleSize, circleSize);
}

Select Menus (Dropdowns):

Code example 4 written in javascript. The code follows.
let mySelect;

function setup() {
  createCanvas(400, 400);
  
  // createSelect()
  mySelect = createSelect();
  mySelect.position(20, 20);
  
  // Add options
  mySelect.option('circle');
  mySelect.option('square');
  mySelect.option('triangle');
  
  // Call function when selection changes
  mySelect.changed(selectionChanged);
}

function draw() {
  background(220);
  
  // Read selected value
  let shape = mySelect.value();  // Returns string
  
  if (shape === 'circle') {
    ellipse(200, 200, 100, 100);
  } else if (shape === 'square') {
    rect(150, 150, 100, 100);
  }
}

function selectionChanged() {
  console.log('Selection changed to: ' + mySelect.value());
}

Controlling Sound Synthesis with UI

UI controls are perfect for manipulating sound synthesis. You can use sliders to control frequency (pitch), amplitude (volume), and other parameters in real-time. This builds on Week 9's sound synthesis lessons by adding user-friendly interfaces.

UI-Controlled Oscillator:

Code example 5 written in javascript. The code follows.
let osc;
let freqSlider;
let ampSlider;
let startButton;

function setup() {
  createCanvas(400, 400);
  
  // Create oscillator
  osc = new p5.Oscillator('sine');
  osc.amp(0.3);
  
  // Frequency slider (pitch control)
  freqSlider = createSlider(100, 1000, 440, 1);
  freqSlider.position(20, 20);
  
  // Amplitude slider (volume control)
  ampSlider = createSlider(0, 100, 30, 1);
  ampSlider.position(20, 50);
  
  // Start button
  startButton = createButton('Start');
  startButton.position(20, 80);
  startButton.mousePressed(startSound);
}

function draw() {
  background(220);
  
  // Update oscillator from sliders
  if (osc && osc.started) {
    let freq = freqSlider.value();
    let amp = ampSlider.value() / 100.0;  // Convert 0-100 to 0.0-1.0
    
    osc.freq(freq);
    osc.amp(amp);
  }
}

function startSound() {
  osc.start();
}

Key Points:

  • Read slider values in draw() for real-time updates
  • Convert slider values to appropriate ranges (e.g., 0-100 to 0.0-1.0 for amplitude)
  • Use buttons for discrete actions (start/stop, change waveform)
  • Use checkboxes to toggle effects (tremolo, reverb, etc.)

Controlling Visual Systems with UI

UI controls can manipulate any visual property: colors, shapes, sizes, positions, animation speeds, and more. This allows for live visual experimentation and interactive installations.

UI-Controlled Visuals:

Code example 6 written in javascript. The code follows.
let redSlider, greenSlider, blueSlider;
let sizeSlider;
let modeButton;
let currentMode = 'circle';

function setup() {
  createCanvas(400, 400);
  
  // Color sliders (RGB: 0-255)
  redSlider = createSlider(0, 255, 255, 1);
  redSlider.position(20, 20);
  
  greenSlider = createSlider(0, 255, 100, 1);
  greenSlider.position(20, 50);
  
  blueSlider = createSlider(0, 255, 100, 1);
  blueSlider.position(20, 80);
  
  // Size slider
  sizeSlider = createSlider(20, 200, 100, 1);
  sizeSlider.position(20, 110);
  
  // Mode button
  modeButton = createButton('Change Mode');
  modeButton.position(20, 140);
  modeButton.mousePressed(changeMode);
}

function draw() {
  background(30);
  
  // Read slider values
  let r = redSlider.value();
  let g = greenSlider.value();
  let b = blueSlider.value();
  let shapeSize = sizeSlider.value();

  // Draw shape with current settings
  fill(r, g, b);
  noStroke();

  if (currentMode === 'circle') {
    ellipse(200, 200, shapeSize, shapeSize);
  } else if (currentMode === 'square') {
    rectMode(CENTER);
    rect(200, 200, shapeSize, shapeSize);
  }
}

function changeMode() {
  if (currentMode === 'circle') {
    currentMode = 'square';
  } else {
    currentMode = 'circle';
  }
}

Audio Input and Video Manipulation

You can capture audio input from a microphone and use it to control video properties. This creates reactive video that responds to sound, perfect for interactive installations and performances.

Audio Input:

p5.js provides p5.AudioIn to capture audio from the microphone. Use getLevel() to get the current volume level (0.0 to 1.0).

Code example 7 written in javascript. The code follows.
let mic;
let audioLevel = 0;

function setup() {
  createCanvas(400, 400);
  
  // Create audio input (microphone)
  mic = new p5.AudioIn();
  mic.start();  // Start capturing audio
  
  // Browser will ask for microphone permission
}

function draw() {
  background(220);
  
  // Get current audio level (0.0 to 1.0)
  audioLevel = mic.getLevel();
  
  // Use audio level to control something
  let circleSize = map(audioLevel, 0, 1, 10, 200);
  ellipse(200, 200, circleSize, circleSize);
  
  // Display audio level
  fill(0);
  text('Audio Level: ' + nf(audioLevel, 1, 3), 20, 30);
}

Video Capture:

Use createCapture(VIDEO) to capture video from the webcam. The browser will ask for camera permission.

Code example 8 written in javascript. The code follows.
let video;

function setup() {
  createCanvas(640, 480);
  
  // Create video capture (webcam)
  video = createCapture(VIDEO);
  video.size(640, 480);
  video.hide();  // Hide the default video element
}

function draw() {
  background(0);
  
  // Display video
  image(video, 0, 0);
}

Audio Input Controlling Video Properties:

Map audio volume to video brightness, contrast, and saturation to create reactive video effects.

Code example 9 written in javascript. The code follows.
let mic;
let video;
let audioLevel = 0;

function setup() {
  createCanvas(640, 480);
  
  // Audio input
  mic = new p5.AudioIn();
  mic.start();
  
  // Video capture
  video = createCapture(VIDEO);
  video.size(640, 480);
  video.hide();
}

function draw() {
  background(0);
  
  // Get audio level
  audioLevel = mic.getLevel();
  
  // Map audio level to video properties
  // Louder = brighter, more contrast, more saturation
  let brightness = 1.0 + (audioLevel * 2.0);  // 1.0 to 3.0
  let contrast = 1.0 + (audioLevel * 1.5);    // 1.0 to 2.5
  let saturation = 1.0 + (audioLevel * 1.5);  // 1.0 to 2.5
  
  // Apply effects to video
  // (Simplified - actual implementation uses pixel manipulation)
  tint(255, saturation * 255);
  image(video, 0, 0);
  noTint();
  
  // Display audio level
  fill(255);
  text('Audio Level: ' + nf(audioLevel, 1, 3), 20, 30);
  text('Brightness: ' + nf(brightness, 1, 2), 20, 50);
  text('Contrast: ' + nf(contrast, 1, 2), 20, 70);
  text('Saturation: ' + nf(saturation, 1, 2), 20, 90);
}

Important Notes:

  • Audio and video capture require browser permissions
  • Works best with HTTPS or localhost (security requirement)
  • getLevel() returns values that can be jittery - consider smoothing
  • Video manipulation can be performance-intensive

Important UI Control Concepts

Control Function Read Value Use Case
createSlider() Continuous values .value() Size, position, color, speed
createButton() Discrete actions N/A (use callback) Start/stop, reset, change mode
createCheckbox() Boolean toggle .checked() Show/hide, enable/disable
createSelect() Multiple choice .value() Choose from options

Common Patterns

Real-Time Parameter Control

Code example 10 written in javascript. The code follows.
let slider;

function setup() {
  createCanvas(400, 400);
  slider = createSlider(0, 255, 128, 1);
}

function draw() {
  // Read slider value every frame for real-time control
  let sliderValue = slider.value();

  // Use value to control something
  fill(sliderValue);
  ellipse(200, 200, 100, 100);
}

Multiple Controls

Code example 11 written in javascript. The code follows.
let rSlider, gSlider, bSlider;
let sizeSlider;
let showCheckbox;

function setup() {
  createCanvas(400, 400);
  
  rSlider = createSlider(0, 255, 255, 1);
  gSlider = createSlider(0, 255, 100, 1);
  bSlider = createSlider(0, 255, 100, 1);
  sizeSlider = createSlider(20, 200, 100, 1);
  showCheckbox = createCheckbox('Show Shape', true);
}

function draw() {
  background(220);
  
  if (showCheckbox.checked()) {
    let r = rSlider.value();
    let g = gSlider.value();
    let b = bSlider.value();
    let circleSize = sizeSlider.value();

    fill(r, g, b);
    ellipse(200, 200, circleSize, circleSize);
  }
}
